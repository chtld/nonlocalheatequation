106  bool ptOutside(util::Point3 p, int o, util::Point3 pb, util::Point3 pt,
107  double theta = 0.0) {
108 
109  if (o == -1) {
110 
111  // straight crack along y-axis
112  return util::compare::definitelyLessThan(p.d_y, pb.d_y) or
113  util::compare::definitelyGreaterThan(p.d_y, pt.d_y);
114 
115  } else if (o == 1) {
116 
117  // straight crack along x-axis ===> pb == pl, pt == pr
118  return util::compare::definitelyLessThan(p.d_x, pb.d_x) or
119  util::compare::definitelyGreaterThan(p.d_x, pt.d_x);
120 
121  } else if (o == 0) {
122 
123  // straight crack at theta angle with x-axis ===> pb == pl, pt == pr
124 
125  // 1. pb ==> (0,0), pt ==> pt - pb, p ==> p - pb
126  // 2. Apply CW rotation to new pt and new p so that crack line
127  // after rotation is simply along x-axis with left point at
128  // origin and right point at transformed new pt
129 
130  util::Point3 pmap = util::transformation::rotateCW2D(p - pb, theta);
131  util::Point3 ptmap = util::transformation::rotateCW2D(pt - pb, theta);
132 
133  return util::compare::definitelyLessThan(pmap.d_x, 0.0) or
134  util::compare::definitelyGreaterThan(pmap.d_x, ptmap.d_x);
135  }
136 
137  return true;
138  }

148 
149  // algorithm is same for any orientation of crack
150  // pt
151  // o
152  // /
153  // p /
154  // o /
155  // /
156  // /
157  // /
158  // o
159  // pb
160  double a = 0.5 * ((pt.d_x - pb.d_x) * (p.d_y - pb.d_y) -
161  (p.d_x - pb.d_x) * (pt.d_y - pb.d_y));
162 
163  // crack is closer to left nodes
164  return !util::compare::definitelyLessThan(a, 0.0);
